191struct adouble {
192 /* Official adouble magic */
193 uint32_t ad_magic;
194 /* Official adouble version number */
195 uint32_t ad_version;
196 char ad_filler[16];
197 struct ad_entry ad_eid[ADEID_MAX];
198 /* the data fork */
199 struct ad_fd ad_data_fork;
200 /* adouble:v2 -> the adouble file
201 * adouble:ea -> the EA fd */
202 struct ad_fd ad_resource_fork;
203 /* adouble:v2 -> ad_resource_fork
204 * adouble:ea -> ad_resource_fork */
205 struct ad_fd *ad_rfp;
206 /* adouble:v2 -> ad_resource_fork
207 * adouble:ea -> ad_data_fork */
208 struct ad_fd *ad_mdp;
209 /* Our adouble version info (AD_VERSION*) */
210 int ad_vers;
211 /* ad_open flags adflags like ADFLAGS_DIR */
212 int ad_adflags;
213 uint32_t ad_inited;
214 int ad_options;
215 /* multiple forks may open one adouble */
216 int ad_refcount;
217 int ad_data_refcount;
218 int ad_meta_refcount;
219 int ad_reso_refcount;
220 /* resource fork len with AFP 3.0
221 * the header parameter size is too small. */
222 off_t ad_rlen;
223 /* mac name (maccharset or UTF8-MAC) */
224 char *ad_name;
225 struct adouble_fops *ad_ops;
226 /* open forks (by others) */
227 uint16_t ad_open_forks;
228 /* Bytes read into ad_data */
229 size_t valid_data_len;
230 char ad_data[AD_DATASZ_MAX];

233#define ADFLAGS_DF (1<<0)
234#define ADFLAGS_RF (1<<1)
235#define ADFLAGS_HF (1<<2)
236#define ADFLAGS_DIR (1<<3)
237#define ADFLAGS_NOHF (1<<4) /* not an error if no metadata fork */
238#define ADFLAGS_NORF (1<<5) /* not an error if no resource fork */
239#define ADFLAGS_CHECK_OF (1<<6) /* check for open forks from us and other afpd's */
240#define ADFLAGS_SETSHRMD (1<<7) /* setting share mode must be done with excl fcnt lock,
241 which implies that the file must be openend rw.
242 If it can't be opened rw (eg EPERM or EROFS) it will
243 be opened ro and the fcntl locks will be shared, that
244 at least prevent other users who have rw access to the
245 file from placing excl locks. */
246#define ADFLAGS_RDWR (1<<8) /* open read/write */
247#define ADFLAGS_RDONLY (1<<9) /* open read only */
248#define ADFLAGS_CREATE (1<<10) /* create file, open called with O_CREAT */
249#define ADFLAGS_EXCL (1<<11) /* exclusive open, open called with O_EXCL */
250#define ADFLAGS_TRUNC (1<<12) /* truncate, open called with O_TRUNC */

333/* AFP attributes, char in `ad ls`, valid for files|dirs */
